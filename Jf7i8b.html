문제

제네릭 메서드를 이용하여 가변 길이 배열을 정렬입니다 이를 해결하는 다음 프로그램에 대해 빈칸을 채우세요 
(strList) carrot 
(intList) 10 20


(strList) carrot 
(intList) 10 20


(strList) carrot 
(intList) 10 20

코드 빈칸 채우기

class MyList<> { 
	private [] array; 
	private int index;  
	private int size;  
	 
	public MyList( int size ) { 
		index = 0;  
		this.size = size;  
		array = new [size]; 
	} 
	 
	public MyList() { 
		this(10); 
	} 
	 
	public void add(  item ) { 
		if( index == size ) { 
			[] newArray = new [ size * 2 ]; 
			for( int i = 0; i < size; i++ ) 
				newArray[i] = array[i]; 
			array = newArray; 
			size = size*2; 
		} 
		array[index++] = item; 
	} 
	 
	public  remove() { 
		if( index == 0 ) 
			return null; 
		return ()array[--index];  
	} 
	 
	public  get( int index ) { 
		return ()array[index]; 
	} 
	 
	public int size() { 
		return index; 
	} 
} 
 
public class GenericClassMain { 
	public static void main(String[] args) { 
		MyList<> strList =  
		                         new MyList<>(2); 
 
		strList.add( "carrot" ); 
		strList.add( "banana" ); 
		strList.remove(); 
		System.out.print( "(strList) " ); 
		for( int i=0; i < strList.size(); i++ ) 
			System.out.print(  
			                  strList.get(i) + " " ); 
 
		MyList<> intList = new MyList<>(2);  
		intList.add( 10 ); 
		intList.add( 20 ); 
		intList.add( 30 ); 
		intList.remove(); 
		System.out.print( "\n(intList) " ); 
		for( int i=0; i < intList.size(); i++ ) 
			System.out.print( intList.get(i) + " " ); 
	} 
}
