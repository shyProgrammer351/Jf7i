문제

파일 원본을 읽어 파일 복사본을 생성하는 파일 복사 문제입니다 이를 해결하는 다음 프로그램에 대해 빈칸을 채우세요 
--------- 파일 원본.txt의 내용 --------- 
파일 복사가 잘 되면 좋겠어요^^
과연, 파일 복사는 잘 될까요?

코드 빈칸 채우기

import java.util.*; 
import java.io.*; 
 
public class FileCopyMain  
{ 
	public static void main( String[] args )  
{	 
		try 
		{ 
			 inFile = new ( new ( new ( "C:\\Users\\user\\Downloads\\원본.txt" ) ) ); 	 
			 outFile = new ( new ( new ( "C:\\Users\\user\\Downloads\\복사본.txt" ) ) ); 	 
			for( String line = ""; ( line = inFile.() ) != null; )  
			{ 
				outFile.( line + "\n" ); 		 
			} 
 
			outFile.();						 
 
			inFile.();							 
			outFile.();						 
		} 
		catch (  e ) 					 
		{ 
			System.out.println( "[오류] 파일을 열 수 없습니다!" ); 
			e.printStackTrace(); 
		} 
 
		return; 
	} 
}
