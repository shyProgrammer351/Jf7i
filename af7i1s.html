문제

char, int, double 형 변수에 !, 15, 3.14의 값을 저장하고 포인터 변수에서 각 변수의 주소값을 
저장하고 기본 자료형 변수와 포인터 변수로 주소값 및 저장값 출력하는 문제입니다 실행순서를 클릭하세요 
변수명   var주소값    저장값
---------------------------------------------
var1     0018FF44     !
var2     0018FF40     15
var3     0018FF38     3.140000
변수명   pnt주소값    var주소값    저장값
---------------------------------------------
pnt1     0018FF34     0018FF44     !
pnt2     0018FF30     0018FF40     15
pnt3     0018FF2C     0018FF38     3.140000
입력1)

프로그램 코드

	#include <stdio.h>

		

1	int main(void)

	{

		char   var1  = '!';	

		int    var2  = 15;	

2		double var3  = 3.14;	


		char *pnt1 = &var1;				

		int *pnt2 = &var2;				

3		double *pnt3 = &var3;				


4		printf("\n변수명   var주소값    저장값\n");

5		printf("---------------------------------------------");

6		printf("var1     %p     %c\n", &var1, var1 );				

7		printf("var2     %p     %d\n", &var2, var2 );				

8		printf("var3     %p     %f\n", &var3, var3 );				


9		printf("\n변수명   pnt주소값    var주소값    저장값\n");

10		printf("---------------------------------------------\n");

11		printf("pnt1     %p     %p     %c\n", &pnt1, pnt1, *pnt1 );		

12		printf("pnt2     %p     %p     %d\n", &pnt2, pnt2, *pnt2 );		

13		printf("pnt3     %p     %p     %f\n", &pnt3, pnt3, *pnt3 );		


14		return 0;

	}
