문제

제네릭 메서드를 이용하여 가변 길이 배열을 정렬하는 문제입니다 실행순서를 클릭하세요 
(strList) carrot 
(intList) 10 20

프로그램 코드

	class MyList {

		private Object[] array;

		private int index; 

		private int size; 

		

1		public MyList( int size ) {

			index = 0; 

			this.size = size; 

			array = new Object[size];

2		}

		

		public MyList() {

			this(10);

		}

		

3		public void add( T item ) {

			if( index == size ) {

				Object[] newArray = new Object[ size * 2 ];

				for( int i = 0; i < size; i++ )

					newArray[i] = array[i];

				array = newArray;

				size = size*2;

			}

			array[index++] = item;

4		}

		

5		public T remove() {

			if( index == 0 )

				return null;

			return (T)array[--index]; 

6		}

		

		public T get( int index ) {

			return (T)array[index];

		}

		

		public int size() {

			return index;

		}

	}
