(strList) carrot

	class MyList {

		private Object[] array;

		private int index; 

		private int size; 

		

1		public MyList( int size ) {

2		}

			this(10);

		}

		

3		public void add( T item ) {

			if( index == size ) {

				Object[] newArray = new Object[ size * 2 ];

				for( int i = 0; i < size; i++ )

					newArray[i] = array[i];

				array = newArray;

				size = size*2;

			}

			array[index++] = item;

4		}

		

5		public T remove() {

			if( index == 0 )

				return null;

			return (T)array[--index]; 

6		}

		

7		public T get( int index ) {

			return (T)array[index];

8		}

		

9		public int size() {

			return index;

10		}
