알고리즘

프로그램 시작 
   먼저 임의의 수 숨기기 
   5번의 기회 내에서 임의의 수 맞추기 
      숫자 입력 받기 
         입력값이 제한범위 벗어나면 다시 입력받기 
      비교 결과 출력 
         첫째 임의의 수를 맞추면 Sucess!! 출력 후 종료 
         둘째 임의의 수가 입력값보다 크면 Up! 출력 
         셋째 임의의 수가 입력값보다 작으면 Down! 출력 
프로그램 종료

프로그램 코드 

	// 파일명 : ./Chapter07/UpAndDown.java

	import java.util.Scanner;

	import java.lang.Math;

		 

	public class UpAndDown

	{

		// 프로그램 시작 

1		public static void main( String[] args ) {

			Scanner scan = new Scanner( System.in );

2			int number = 0, user = 0, chance = 6;

		 

			// 먼저 임의의 수 숨기기 

3			number = (int)( Math.random() * 100 ) + 1;

		 

			// 5번의 기회 내에서 임의의 수 맞추기 

4			while( --chance > 0 ) {

5				System.out.println( "기회는 " + chance + "번 남았습니다" );

		 

				// 숫자 입력 받기 

6				System.out.print( "1~100사이의 숫자를 입력하세요: " );

7				user = scan.nextInt();

		 

				// 입력값이 제한범위 벗어나면 다시 입력받기 

8				if ( ( user < 1 ) || ( 100 < user ) ) {

9					System.out.println( "제한범위를 벗어났습니다" );

10					chance++;

11					continue;

				}

				// 비교 결과 출력 

				// 첫째 임의의 수를 맞추면 Sucess!! 출력 후 종료 

12				else if ( number == user ) {

13					System.out.println( "Success!!" );

14					break;

				}

				// 둘째 임의의 수가 입력값보다 크면 Up! 출력 

15				else if ( number > user ) {

16					System.out.println( "Up!" );

				}

				// 셋째 임의의 수가 입력값보다 작으면 Down! 출력 

17				else if ( number < user ) {

18					System.out.println( "Down!" );

				}

			}

		 

		// 프로그램 종료 

19		}

	}
