코드 빈칸 채우기

import java.util.Scanner; 
	 
public class CharacterMovement 
{ 
	public static void main( String[] args ) { 
		Scanner scan = new Scanner( System.in ); 
		int row = 0, column = 9; 
		char direction='w'; 
	 
		System.out.println( " \033[44m                  \033[0m " ); 
		( row < 8 ) 
			System.out.println( " \033[44m  \033[0m                \033[44m  \033[0m " ); 
		System.out.println( " \033[44m                  \033[0m " ); 
		 
		 { 
			System.out.print( " \033[" + row + ";" + column + "f  " ); 
			switch( direction ) { 
				case 'w' case 'W'  
					row = ( row <= 2 )  2  row-1; 
					break; 
				case 'a' case 'A'  
					column = ( column <= 2 )  2  column-1; 
					break; 
				case 's' case 'S'  
					row = ( row >= 8 )  8  row+1; 
					break; 
				case 'd' case 'D'  
					column = ( column >= 15 )  15  column+1; 
					break; 
			} 
			System.out.print( " \033[" + row + ";" + column + "f옷" ); 
	 
			System.out.print( " \033[10;1f \033[2KWASD와 [Enter]를 입력하세요: " ); 
			direction= scan.nextLine().charAt(0); 
		} ( ( direction  'W' )  ( direction  'A' )  ( direction  'S' )  ( direction  'D' ) 
		         ( direction  'w' )  ( direction  'a' )  ( direction  's' )  ( direction  'd' ) ); 
	 
	} 
}
