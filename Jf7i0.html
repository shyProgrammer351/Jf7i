7. 반복문(while)

while( 조건식 ) {
  조건식이 참이면 실행하는 명령문
} 		do {
  조건식이 참이면 실행하는 명령문
} while( 조건식 );

while   조건식이 참일 때마다 해당 명령문을 실행하는 반복문
조건식(참) → 명령문 → 조건식(참) → 명령문 → ... → 조건식(거짓) → while문 종료의 순서대로 실행

do   한 번 이상 실행하는 반복문의 시작을 표현
명령문 → 조건식(참) → 명령문 → 조건식(참) → ... → 조건식(거짓) → do-while문 종료의 순서대로 실행

while( 조건식 ) {
  조건식이 참이면 실행하는 명령문
  break;
  break문이 적용되지 않을 때만 실행하는 명령문
} 		while( 조건식 ) {
  조건식이 참이면 실행하는 명령문
  continue;
  continue가 적용되지 않을 때만 실행하는 명령문
}

break   실행흐름을 중단하고 해당 영역을 빠져나오는 명령문
while문, for문, switch문에서 break를 실행하면 이후 이후 명령문이나 조건식은 모두 건너띄고 반복문 중단

continue   이후의 명령문은 실행하지 않고 반복문의 조건식으로 실행흐름을 바꾸는 명령문
조건식(참) → 명령문 → continue → 조건식(참) → 명령문 → ...의 순서대로 실행

import java.util.Scanner;
import java.lang.Math;

public class Main {
  public static void main( String[] args ) {
    Scanner scan = new Scanner( System.in );
    int number = (int)( Math.random() * 100 ) + 1;
    int chance = 5;
    while( --chance > 0 ) {
      System.out.print( "1~100사이의 숫자를 입력하세요: " );
      int user = scan.nextInt();
      if ( ( user < 1 ) || ( 100 < user ) ) {
        System.out.println( "제한범위를 벗어났습니다" );
        chance++;
        continue;   while(조건식)으로 이동
      }
      else if ( number == user ) {
        System.out.println( "Success!!" );
        break;   while문 종료 후 main 메서드 끝으로 이동
      }
      else if ( number > user ) {
        System.out.println( "Up!" );
      }
      else if ( number < user ) {
        System.out.println( "Down!" );
      }
      System.out.println( "기회는 " + chance + "번 남았습니다" );
    }
  }   main 메서드 끝
}
