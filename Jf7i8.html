프로그램 코드 

	// 파일명 : ./Chapter07/GenericClassMain.java 

	class MyList<T> { 

		private Object[] array; 

		private int index;  

		private int size;  

		 

1		public MyList( int size ) { 

			index = 0;  

			this.size = size;  

			array = new Object[size]; 

2		} 

		 

		public MyList() { 

			this(10); 

		} 

		 

3		public void add( T item ) { 

			if( index == size ) { 

				Object[] newArray = new Object[ size * 2 ]; 

				for( int i = 0; i < size; i++ ) 

					newArray[i] = array[i]; 

				array = newArray; 

				size = size*2; 

			} 

			array[index++] = item; 

4		} 

		 

5		public T remove() { 

			if( index == 0 ) 

				return null; 

			return (T)array[--index];  

6		} 

		 

7		public T get( int index ) { 

			return (T)array[index]; 

8		} 

		 

9		public int size() { 

			return index; 

10		} 

	}

16			for( int i=0; i < strList.size(); i++ )

18				                  strList.get(i) + " " );

			// MyList에 정수 제거

			// MyList를 출력
