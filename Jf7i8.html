문제

제네릭 메서드를 이용하여 가변 길이 배열을 정렬하는 문제입니다 이를 해결하는 다음 프로그램을 해석하세요 
(strList) carrot 
(intList) 10 20


(strList) carrot 
(intList) 10 20


(strList) carrot 
(intList) 10 20

	// 파일명 : ./Chapter07/GenericClassMain.java 

	// 다양한 자료형 데이터를 관리하는 리스트 클래스 

	class MyList<T> { 

		private Object[] array; 

		private int index;  

		private int size;  

		 

		// MyList 생성 초기화  

1		public MyList( int size ) { 

			index = 0;  

			this.size = size;  

			array = new Object[size]; 

2		} 

		 

		public MyList() { 

			this(10); 

		} 

		 

		// MyList에 항목 추가   

3		public void add( T item ) { 

			// 더이상 항목을 추가할 공간이 없으면 기존의 2배 공간 확보   

			if( index == size ) { 

				Object[] newArray = new Object[ size * 2 ]; 

				for( int i = 0; i < size; i++ ) 

					newArray[i] = array[i]; 

				array = newArray; 

				size = size*2; 

			} 

			// 항목 추가   

			array[index++] = item; 

4		} 

		 

		// MyList에 항목 삭제  

5		public T remove() { 

			if( index == 0 ) 

				return null; 

			return (T)array[--index];  

6		} 

		 

		// MyList에서 index 항목 반환  

		public T get( int index ) { 

			return (T)array[index]; 

		} 

		 

		// MyList의 크기 반환    

		public int size() { 

			return index; 

		} 

	}
